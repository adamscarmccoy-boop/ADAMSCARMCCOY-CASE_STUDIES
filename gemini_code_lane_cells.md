  "if n_docs < auto_k:\n",
  "    auto_k = max(1, n_docs // 2)\n",
  "    if auto_k == 0: auto_k = 1 # Ensure at least one cluster if data exists\n",

  "    df_blocks['cluster'] = kmeans_model.fit_predict(tfidf_matrix)\n",
  "    print(f\"K-Means clustering complete with K={auto_k} clusters.\")\n",
  "\n",
  "    # Print cluster taxonomy\n",
  "    terms = tfidf_vectorizer.get_feature_names_out()\n",
  "    centroids = kmeans_model.cluster_centers_.argsort()[:, ::-1]\n",
  "    print(f\"\\n--- Your Code Taxonomy ({auto_k} clusters): ---\")\n",
  "    for i in range(auto_k):\n",
  "        cluster_rows = df_blocks[df_blocks['cluster'] == i]\n",
  "        top_terms = [terms[j] for j in centroids[i, :6]]\n",
  "        print(f\"  Cluster {i:>2} ({len(cluster_rows):>4} blocks): {', '.join(top_terms)}\")\n",
  "else:\n",
  "    df_blocks['cluster'] = -1 # No clustering if not enough data\n",

  "    if any(w in combined for w in ['model', 'schema', 'pydantic', 'basemodel', 'data_structure']): return 'schema'\n",
  "    if any(w in combined for w in ['audit', 'verify', 'check', 'validate', 'assert']): return 'audit'\n",
  "    if any(w in combined for w in ['deploy', 'docker', 'cloud', 'gcs', 'firebase', 'kubernetes']): return 'deployment'\n",
  "    if any(w in combined for w in ['sovereign', 'autonomous', 'swarm', 'agent', 'bot']): return 'agent'\n",
  "    if row['type'] == 'class': return 'class_def'\n",
  "    return 'utility'\n",
